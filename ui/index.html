        function addLine(html, color) {

            const line = document.createElement('div');

            line.className = color;

            line.innerHTML = html;

            output.appendChild(line);

            output.scrollTop = output.scrollHeight;

        }


        function addLinesSequentially(lines, container, finalCallback) {

            let index = 0;

            function next() {

                if (index < lines.length) {

                    const lineData = lines[index];

                    const lineEl = document.createElement('div');

                    lineEl.className = lineData.color;

                    lineEl.innerHTML = lineData.text;

                    container.appendChild(lineEl);

                    container.scrollTop = container.scrollHeight;

                    index++;

                    setTimeout(next, 100 + Math.random() * 150);

                } else {

                    if (finalCallback) finalCallback();

                }

            }

            next();

        }


        function processCommand(cmd) {

            const parts = cmd.toLowerCase().split(' ').filter(p => p);

            const command = parts[0];

            const args = parts.slice(1);


            switch (command) {

                case '/clear':

                    output.innerHTML = '';

                    enableInput();

                    break;

                case '/help':

                    addLine(knowledgeBase.help, 'text-cyan-400');

                    enableInput();

                    break;

                case '/status':

                     addLine(knowledgeBase.status(), 'text-cyan-400');

                     enableInput();

                    break;

                case '/ask':

                    const topic = args.includes('about') ? args[args.indexOf('about') + 1] : args[0];

                    if (!topic) {

                        addLine("Usage: /ask [topic]", 'text-amber-400');

                        enableInput();

                        return;

                    }

                    const response = knowledgeBase[topic] || knowledgeBase.default;

                    addLine(response, 'text-cyan-400');

                    enableInput();

                    break;

                case '/analyze':

                    const riskType = args[0];

                    if (analysisTasks[riskType]) {

                        addLinesSequentially(analysisTasks[riskType], output, enableInput);

                    } else {

                        addLine("Unknown analysis type. Try 'centralization' or 'spam'.", 'text-amber-400');

                        enableInput();

                    }

                    break;

                case '/simulate':

                     if (args[0] === 'cycle') {

                        addLinesSequentially(analysisTasks.simulation, output, enableInput);

                     } else {

                        addLine("Unknown simulation. Try '/simulate cycle'.", 'text-amber-400');

                        enableInput();

                     }

                    break;

                default:

                    addLine(`Command not found: ${cmd}. Type '/help' for available commands.`, 'text-amber-400');

                    enableInput();

                    break;

            }

        }


        function disableInput() {

            input.disabled = true;

            input.style.cursor = 'not-allowed';

        }


        function enableInput() {

            inputLine.style.display = 'flex';

            input.disabled = false;

            input.style.cursor = 'text';

            input.focus();

        }


        input.addEventListener('keydown', (e) => {

            if (e.key === 'Enter') {

                e.preventDefault();

                const cmd = input.value;

                if (cmd.trim() === '') return;


                commandHistory.unshift(cmd);

                historyIndex = -1;

                

                const historyLine = document.createElement('div');

                historyLine.innerHTML = `<span class="text-cyan-400">&gt;</span> <span class="text-magenta-400">${cmd}</span>`;

                output.appendChild(historyLine);

                

                input.value = '';

                autocompleteSuggestionEl.textContent = '';

                disableInput();

                

                setTimeout(() => processCommand(cmd), 300);

            } else if (e.key === 'Escape') {

                input.value = '';

                autocompleteSuggestionEl.textContent = '';

            } else if (e.key === 'ArrowUp') {

                e.preventDefault();

                if (historyIndex < commandHistory.length - 1) {

                    historyIndex++;

                    input.value = commandHistory[historyIndex];

                }

            } else if (e.key === 'ArrowDown') {

                e.preventDefault();

                if (historyIndex > 0) {

                    historyIndex--;

                    input.value = commandHistory[historyIndex];

                } else {

                    historyIndex = -1;

                    input.value = '';

                }

            } else if (e.key === 'Tab') {

                e.preventDefault();

                if (autocompleteSuggestionEl.textContent) {

                    input.value = autocompleteSuggestionEl.textContent;

                    autocompleteSuggestionEl.textContent = '';

                }

            }

        });


        const allCommands = ['/ask', '/analyze', '/simulate', '/clear', '/status', '/help'];

        input.addEventListener('input', () => {

            const currentVal = input.value;

            if (currentVal.startsWith('/')) {

                const suggestion = allCommands.find(c => c.startsWith(currentVal));

                if (suggestion && suggestion !== currentVal) {

                    autocompleteSuggestionEl.textContent = suggestion;

                } else {

                    autocompleteSuggestionEl.textContent = '';

                }

            } else {

                autocompleteSuggestionEl.textContent = '';

            }

        });
